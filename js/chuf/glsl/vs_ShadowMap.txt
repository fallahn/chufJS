//////////////////////////////////////
// vertex shader for shadow map render
//////////////////////////////////////

//GLES 1.0 / WebGL

#version 100

attribute vec3 aPosition;

uniform mat4 uPMat;
uniform mat4 uMMat;
uniform mat4 uVMat;

void main()
{
	gl_Position = uPMat * uVMat * uMMat * vec4(aPosition, 1.0);
}

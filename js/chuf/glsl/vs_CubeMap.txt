///////////////////////////////////////////////////
// vertex shader for passing coordinates of cubemap
///////////////////////////////////////////////////

//GLES 2.0 / WebGL

#version 100

attribute vec3 aPosition;

uniform mat4 uPMat;
uniform mat4 uMVMat;
uniform mat4 uCMat;

varying vec3 vPosition;

void main()
{
	vec4 viewVert = uCMat * uMVMat * vec4(aPosition, 1.0);
	gl_Position = uPMat * viewVert;
	vPosition = aPosition;
}

//////////////////////////////////////////////////////////
// vertex shader for drawing coloured mesh debug output //
//////////////////////////////////////////////////////////

//GLES 2.0 / WebGL

#version 100

attribute vec3 aPosition;
attribute vec4 aColour;

varying vec4 vColour;

uniform mat4 uMVMat;
uniform mat4 uPMat;

void main()
{
	gl_Position = uPMat * uMVMat * vec4(aPosition, 1.0);
	vColour = aColour;
}

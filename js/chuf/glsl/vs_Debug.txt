//////////////////////////////////////////////////////////
// vertex shader for drawing coloured mesh debug output //
//////////////////////////////////////////////////////////

//GLES 1.0 / WebGL

#version 100

attribute vec3 aPosition;
attribute vec4 aColour;

varying vec4 vColour;

uniform mat4 uMMat;
uniform mat4 uVMat;
uniform mat4 uPMat;

void main()
{
	gl_Position = uPMat * uVMat * uMMat * vec4(aPosition, 1.0);
	vColour = aColour;
}
